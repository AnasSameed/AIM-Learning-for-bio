Code for today
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# NumPy — fast array math
arr = np.array([1, 2, 3, 4, 5])
print(arr * 2)         # [2 4 6 8 10]
print(arr.mean())      # 3.0
print(arr.std())       # 1.414

# Matrix operations (critical for ML)
A = np.array([[1,2],[3,4]])
B = np.array([[5,6],[7,8]])
print(np.dot(A, B))    # matrix multiply

# Pandas — load and explore data
df = pd.read_csv('data.csv')
print(df.head())
print(df.describe())   # stats summary
print(df.isnull().sum()) # check missing

# Matplotlib — visualize
plt.figure(figsize=(8,4))
plt.hist(df['column'], bins=30)
plt.title('Distribution')
plt.show()
